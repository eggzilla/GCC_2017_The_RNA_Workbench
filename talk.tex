  \frametitle{Goals of the RNA Workbench}
  %Differences to previous work
  \begin{figure}
    \centering
    \includegraphics[width=0.25\textwidth]{figures/welcome_image.eps}
  \end{figure}
  \begin{itemize}
  \item Comprehensive set of RNA-bioinformatics tools
  \item Easy and stable dissemination via Galaxy and Docker
  \item Set of predefined workflows and associated descriptions/training material
  \item Community - authors and users
  \end{itemize}
  
}

\frame{
  \frametitle{Tools}
  \begin{itemize}
  \item More than 50 RNA bioinformatics tools
  \item Classical RNA bioinformatics (INFERNAL, ViennaRNA, ..)
  \item RNA-seq analysis (FuMa, Paralizer, ..)
  \item Auxiliary tools - Visualisation
  \end{itemize}  
}

  \item Template for customized workflows
  \end{itemize}
